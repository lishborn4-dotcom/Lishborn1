    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    
    <!-- Load Babel for JSX support -->
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <style>
        /* Custom Font Import if needed, using system mono for terminal feel */
        body {
            font-family: 'Menlo', 'Monaco', 'Courier New', monospace;
            background-color: #101010;
        }
        
        @keyframes bounce-slow {
            0%, 100% { transform: scale(1.25) translateY(0); }
            50% { transform: scale(1.25) translateY(-5px); }
        }
        .animate-bounce-slow {
            animation: bounce-slow 3s infinite;
        }

<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect } = React;

        // --- รายการคำด่า (แก้ไขตรงนี้ได้เลย) ---
        const GENERATED_RESPONSES = [
            "ออกแบบให้แย่ยังไงก็ได้เพราะยังไงก็ไม่แย่กว่าแม่มแล้ว",
            "ระบบล่มเก่งกว่าระบบทำงานอีกครับ",
            "งบประมาณพันล้าน ได้งานหลักร้อย",
            "บัตรเขย่งหรือคนเขย่งครับเนี่ย",
            "ขอเครื่องคิดเลขใหม่ไหมครับ?",
            "Loading 99%... (ค้างมา 4 ปีแล้ว)",
            "Error 404: Transparency Not Found",
            "ต้องใช้ AI ช่วยทำงาน หรือต้องใช้คนมี I ทำงานครับ?",
            "กกต.หน้าหี",
            "กกต.หัวควย",
            "ไอเหี้ย กกต.",
            "กกต.ส้นตีน",
            "กกต.จัญไร"
        ];

        const PixelGhost = () => (
            <svg width="64" height="64" viewBox="0 0 14 14" fill="none" xmlns="http://www.w3.org/2000/svg" className="text-red-500">
                <path d="M4 2H10V3H11V4H12V9H11V10H10V11H9V10H8V9H6V10H5V11H4V10H3V9H2V4H3V3H4V2Z" fill="currentColor"/>
                <rect x="4" y="5" width="2" height="2" fill="black"/>
                <rect x="8" y="5" width="2" height="2" fill="black"/>
                <rect x="4" y="6" width="1" height="1" fill="white"/>
                <rect x="8" y="6" width="1" height="1" fill="white"/>
            </svg>
        );

        function App() {
            const [displayedText, setDisplayedText] = useState("");
            const [isTyping, setIsTyping] = useState(false);
            const [cursorVisible, setCursorVisible] = useState(true);
            
            const generateNewInsult = () => {
                if (isTyping) return;
                
                setIsTyping(true);
                setDisplayedText("");
                
                const randomIndex = Math.floor(Math.random() * GENERATED_RESPONSES.length);
                const targetText = GENERATED_RESPONSES[randomIndex];
                
                let charIndex = 0;
                
                const typingInterval = setInterval(() => {
                    if (charIndex < targetText.length) {
                        setDisplayedText((prev) => prev + targetText.charAt(charIndex));
                        charIndex++;
                    } else {
                        clearInterval(typingInterval);
                        setIsTyping(false);
                    }
                }, 50);
            };

            useEffect(() => {
                const interval = setInterval(() => setCursorVisible(v => !v), 530);
                return () => clearInterval(interval);
            }, []);

            useEffect(() => {
                generateNewInsult();
            }, []);

            return (
                <div className="min-h-screen text-gray-300 flex items-center justify-center p-4 selection:bg-gray-700 selection:text-white"
                     onClick={generateNewInsult}>
                    <div className="w-full max-w-2xl cursor-pointer">
                        
                        {/* Main Red Box */}
                        <div className="relative border border-red-500/40 rounded-xl pt-16 pb-12 px-8 flex flex-col items-center justify-center gap-6 bg-[#101010] transition-colors hover:border-red-500/60">
                            <div className="absolute -top-3 left-8 px-2 bg-[#101010]">
                                <span className="text-red-400/80 text-sm tracking-wide">Claude Code</span>
                            </div>

                            <h1 className="text-xl text-gray-200 font-bold tracking-tight">Welcome back</h1>

                            <div className="transform scale-125 my-2 animate-bounce-slow">
                                <PixelGhost />
                            </div>

                            <div className="text-center space-y-1">
                                <p className="text-gray-400 font-medium">Opus 4.5</p>
                                <p className="text-gray-500 text-sm">API Usage Billing</p>
                            </div>
                        </div>

                        {/* Terminal Input Area */}
                        <div className="mt-10 px-2 space-y-3">
                            <div className="text-gray-500 text-sm mb-4">Welcome to Opus 4.5</div>
                            <div className="h-px w-full bg-gray-800 mb-4"></div>

                            {/* User Command */}
                            <div className="flex items-center gap-3 text-lg">
                                <span className="text-gray-500 font-bold">❯</span>
                                <span className="text-gray-200">สร้างระบบด่ากกต.อัตโนมัติ</span>
                            </div>

                            {/* AI Response */}
                            <div className="text-lg text-gray-400 pl-6 relative min-h-[30px]">
                                <span className={`inline-block w-2.5 h-6 bg-gray-400 align-middle mr-1 transition-opacity ${cursorVisible || isTyping ? 'opacity-100' : 'opacity-0'}`}></span>
                                <span>{displayedText}</span>
                            </div>
                        </div>

                        {/* Footer */}
                        <div className="mt-16 px-2 flex flex-col items-center gap-2 text-cyan-600/60 text-xs font-medium opacity-70">
                            <div>(Click anywhere to generate new response)</div>
                            <div className="flex items-center gap-2">
                                <span>❚❚</span>
                                <span>plan mode on (shift+tab to cycle)</span>
                            </div>
                        </div>

                    </div>
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
